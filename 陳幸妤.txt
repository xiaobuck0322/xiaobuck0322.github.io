# BMI 公式：BMI = 體重(公斤) / (身高(公尺))^2

height = float(input("請輸入身高（公尺）："))  # 例如：1.7
weight = float(input("請輸入體重（公斤）："))  # 例如：60

bmi = weight / (height ** 2)
print("你的 BMI 是：", round(bmi, 2))

# 健康建議
if bmi < 18.5:
    print("體重過輕")
elif bmi < 24:
    print("正常範圍")
elif bmi < 27:
    print("稍重")
else:
    print("可能過重，注意健康")
